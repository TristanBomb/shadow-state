def initialize(tag, world:, flag: nil, color: "#000000", gov_type: :monarchy)
  if world.countries.select{|c| c.tag == tag}.any?
    raise "country tag #{tag} exists already!"
  end
  @tag = tag
  @flag = flag
  @color = color
  super(world)
end
